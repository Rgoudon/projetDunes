[docs]	def coordonnees(self, x, y):
		"""
		Fonction de verification des coordonnées d'un coin.
		
		:param x: coordonnée en abscisse du coin à ajouter.
		:type x: int
		
		:param y: coordonnée en ordonnée du coin à ajouter.
		:type y: int
		
		"""
		if(self.bX == False and self.bY == False):
			self.setValues(x, y)
			return x, y

		Xi = x-self.x
		Yi = y-self.y
		if(abs(Xi)>abs(Yi)):
			if(self.bX):
				self.bX = False
				self.x += Xi
			elif(self.bY):
				self.bY = False
				self.y += Yi
		else:
			if(self.bY):
				self.bY = False
				self.y += Yi
			elif(self.bX):
				self.bX = False
				self.x += Xi

		return self.x, self.y
